39/// Get period statistics.
40pub async fn stats(
41    State(state): State<AppState>,
42    Query(params): Query<PeriodParams>,
43) -> Json<Value> {
44    match read::get_stats(&state.db, params.days) {
45        Ok(s) => Json(json!({
46            "period_days": params.days,
47            "spend_usd": s.spend_usd,
48            "requests": s.requests,
49            "input_tokens": s.input_tokens,
50            "output_tokens": s.output_tokens,
51            "prev_period_spend_usd": s.prev_period_spend_usd,
52            "avg_cost_per_request": if s.requests > 0 { s.spend_usd / s.requests as f64 } else { 0.0 },
53        })),
54        Err(e) => Json(json!({ "error": e.to_string() })),
55    }
56}

102/// Get AI-generated insights.
103pub async fn insights(
104    State(state): State<AppState>,
105    Query(params): Query<PeriodParams>,
106) -> Json<Value> {
107    let stats = match read::get_stats(&state.db, params.days) {
108        Ok(s) => s,
109        Err(e) => return Json(json!({"error": e.to_string()})),
110    };
111    let models = match read::get_model_breakdown(&state.db, params.days) {
112        Ok(m) => m,
113        Err(e) => return Json(json!({"error": e.to_string()})),
114    };
115    let tasks = match read::get_task_breakdown(&state.db, params.days) {
116        Ok(t) => t,
117        Err(e) => return Json(json!({"error": e.to_string()})),
118    };
119    let cache = match read::get_cache_stats(&state.db, params.days) {
120        Ok(c) => c,
121        Err(e) => return Json(json!({"error": e.to_string()})),
122    };
123    let anomalies = match read::get_anomaly_stats(&state.db, params.days) {
124        Ok(a) => a,
125        Err(e) => return Json(json!({"error": e.to_string()})),
126    };
127
128    let output =
129        insights_agent::generate_insights(&stats, &models, &tasks, &cache, &anomalies, params.days);
130    Json(serde_json::to_value(output).unwrap_or(json!({"error": "serialization failed"})))
131}
