20/// Main proxy handler — catches all requests to /:provider/*path
21pub async fn proxy_handler(
22    State(state): State<AppState>,
23    Path((provider, path)): Path<(String, String)>,
24    method: Method,
25    headers: HeaderMap,
26    body: Bytes,
27) -> impl IntoResponse {
28    let start = std::time::Instant::now();
29    let request_id = Uuid::new_v4().to_string();
30
31    tracing::info!("[{}] {} /{}/{}", request_id, method, provider, path);
32
33    // 1. Resolve provider base URL
34    let base_url = match state.config.get_base_url(&provider) {
35        Some(url) => url,
36        None => {
37            tracing::error!("Unknown provider: {}", provider);
38            return Response::builder()
39                .status(StatusCode::BAD_REQUEST)
40                .body(Body::from(format!(
41                    "{{\"error\": \"Unknown provider: {}\"}}",
42                    provider
43                )))
44                .unwrap();
45        }
46    };
47
48    // 2. Get API key from environment
49    let api_key = state.config.get_api_key(&provider);
50
51    // 3. Parse request body to extract model info
52    let body_json: Option<serde_json::Value> = serde_json::from_slice(&body).ok();
53    let original_model = body_json
54        .as_ref()
55        .and_then(|b| b.get("model"))
56        .and_then(|m| m.as_str())
57        .unwrap_or("unknown")
58        .to_string();
59
60    // 4. Extract last user message for routing/classification
61    let last_user_message = extract_last_user_message(&body_json);
62
63    // 5. Run Router Agent (rule-based) to decide routing
64    let route_decision = router_agent::decide_route(
65        &state.config,
66        &provider,
67        &original_model,
68        &last_user_message,
69    );
70
71    // 6. Check for block decision
72    if route_decision.action == "block" {
73        tracing::warn!(
74            "[{}] Request blocked: {}",
75            request_id,
76            route_decision.block_reason.as_deref().unwrap_or("unknown")
77        );
78        return Response::builder()
79            .status(StatusCode::TOO_MANY_REQUESTS)
80            .body(Body::from(serde_json::json!({
81                "error": {
82                    "message": route_decision.block_reason.unwrap_or("Request blocked by cost policy".to_string()),
83                    "type": "cost_limit_exceeded"
84                }
85            }).to_string()))
86            .unwrap();
87    }
88
89    // 7. Apply substitution if needed
90    let (effective_provider, effective_model) = if route_decision.action == "substitute" {
91        tracing::info!(
92            "[{}] Substituting {} -> {} ({})",
93            request_id,
94            original_model,
95            route_decision.model,
96            route_decision.substitution_reason.as_deref().unwrap_or("")
97        );
98        (
99            route_decision.provider.clone(),
100            route_decision.model.clone(),
101        )
102    } else {
103        (provider.clone(), original_model.clone())
104    };
105
106    let was_substituted = route_decision.action == "substitute";
107
108    // 8. Modify body if model was substituted
109    let final_body = if was_substituted {
110        if let Some(mut json) = body_json.clone() {
111            json["model"] = serde_json::Value::String(effective_model.clone());
112            Bytes::from(serde_json::to_vec(&json).unwrap_or_else(|_| body.to_vec()))
113        } else {
114            body.clone()
115        }
116    } else {
117        body.clone()
118    };
119
120    // 9. Build target URL
121    let effective_base = state
122        .config
123        .get_base_url(&effective_provider)
124        .unwrap_or(base_url);
125    let target_url = forwarder::build_target_url(&effective_base, &path);
126
127    // 10. Build headers map
128    let mut header_map: HashMap<String, String> = HashMap::new();
129    for (key, value) in headers.iter() {
130        if let Ok(v) = value.to_str() {
131            header_map.insert(key.to_string(), v.to_string());
132        }
133    }
134
135    // Inject API key if available and not already present
136    if let Some(key) = &api_key {
137        if provider == "anthropic" {
138            header_map
139                .entry("x-api-key".to_string())
140                .or_insert_with(|| key.clone());
141        } else {
142            header_map
143                .entry("authorization".to_string())
144                .or_insert_with(|| format!("Bearer {}", key));
145        }
146    }
147
148    // 11. Forward request
149    let client = &state.http_client;
150    let response = match forwarder::forward_request(
151        client,
152        &target_url,
153        method.as_str(),
154        &header_map,
155        final_body,
156    )
157    .await
158    {
159        Ok(resp) => resp,
160        Err(e) => {
161            tracing::error!("[{}] Forward error: {}", request_id, e);
162            return Response::builder()
163                .status(StatusCode::BAD_GATEWAY)
164                .body(Body::from(format!(
165                    "{{\"error\": \"Proxy forward error: {}\"}}",
166                    e
167                )))
168                .unwrap();
169        }
170    };
171
172    let latency_ms = start.elapsed().as_millis() as i64;
173    let response_status = response.status();
174
175    // 12. Copy response headers
176    let mut resp_builder = Response::builder().status(response_status.as_u16());
177    for (key, value) in response.headers() {
178        if key != "transfer-encoding" {
179            resp_builder = resp_builder.header(key, value);
180        }
181    }
182
183    // 13. Check if streaming
184    let is_stream = forwarder::is_sse_response(&response);
185
186    if is_stream {
187        // For streaming responses, pass through and log asynchronously
188        let db = state.db.clone();
189        let pricing_table = state.pricing.clone();
190        let prov = effective_provider.clone();
191        let model = effective_model.clone();
192        let orig_model = original_model.clone();
193        let anomaly_mult = state.config.alerts.anomaly_multiplier;
194        let user_msg = last_user_message.clone();
195
196        // Create a streaming passthrough that captures the final usage chunk
197        let (tx, rx) = tokio::sync::mpsc::channel::<Result<Bytes, std::io::Error>>(64);
198        let mut byte_stream = response.bytes_stream();
199
200        tokio::spawn(async move {
201            use futures_util::StreamExt;
202            let mut all_chunks: Vec<Bytes> = Vec::new();
203
204            while let Some(chunk_result) = byte_stream.next().await {
205                match chunk_result {
206                    Ok(chunk) => {
207                        all_chunks.push(chunk.clone());
208                        if tx.send(Ok(chunk)).await.is_err() {
209                            break;
210                        }
211                    }
212                    Err(e) => {
213                        let _ = tx.send(Err(std::io::Error::other(e.to_string()))).await;
214                        break;
215                    }
216                }
217            }
218            drop(tx);
219
220            // Extract usage from the accumulated SSE chunks
221            if let Some(final_data) =
222                crate::proxy::streamer::extract_usage_from_sse_chunks(&all_chunks)
223            {
224                let usage = normalizer::normalize_response(&prov, &final_data);
225                log_request(
226                    &db,
227                    &pricing_table,
228                    &request_id,
229                    &prov,
230                    &model,
231                    &orig_model,
232                    was_substituted,
233                    &usage,
234                    latency_ms,
235                    anomaly_mult,
236                    &user_msg,
237                )
238                .await;
239            }
240        });
241
242        let body_stream = tokio_stream::wrappers::ReceiverStream::new(rx);
243        let body = Body::from_stream(body_stream);
244        resp_builder.body(body).unwrap()
245    } else {
246        // Non-streaming: read full body, extract usage, log, return
247        let resp_bytes = match response.bytes().await {
248            Ok(b) => b,
249            Err(e) => {
250                tracing::error!("[{}] Failed to read response body: {}", request_id, e);
251                return Response::builder()
252                    .status(StatusCode::BAD_GATEWAY)
253                    .body(Body::from(format!(
254                        "{{\"error\": \"Failed to read response: {}\"}}",
255                        e
256                    )))
257                    .unwrap();
258            }
259        };
260
261        // Parse and extract usage from response
262        if let Ok(resp_json) = serde_json::from_slice::<serde_json::Value>(&resp_bytes) {
263            let usage = normalizer::normalize_response(&effective_provider, &resp_json);
264
265            // Log asynchronously
266            let db = state.db.clone();
267            let pricing_table = state.pricing.clone();
268            let prov = effective_provider.clone();
269            let model = effective_model.clone();
270            let orig_model = original_model.clone();
271            let anomaly_mult = state.config.alerts.anomaly_multiplier;
272            let user_msg = last_user_message.clone();
273            let rid = request_id.clone();
274
275            tokio::spawn(async move {
276                log_request(
277                    &db,
278                    &pricing_table,
279                    &rid,
280                    &prov,
281                    &model,
282                    &orig_model,
283                    was_substituted,
284                    &usage,
285                    latency_ms,
286                    anomaly_mult,
287                    &user_msg,
288                )
289                .await;
290            });
291        }
292
293        resp_builder.body(Body::from(resp_bytes)).unwrap()
294    }
295}

297/// Extract the last user message from the request body for routing/classification.
298fn extract_last_user_message(body: &Option<serde_json::Value>) -> String {
299    body.as_ref()
300        .and_then(|b| b.get("messages"))
301        .and_then(|m| m.as_array())
302        .and_then(|arr| {
303            arr.iter()
304                .rev()
305                .find(|msg| msg.get("role").and_then(|r| r.as_str()) == Some("user"))
306        })
307        .and_then(|msg| msg.get("content"))
308        .and_then(|c| {
309            // Content can be a string or array of content blocks
310            if let Some(s) = c.as_str() {
311                Some(s.to_string())
312            } else if let Some(arr) = c.as_array() {
313                arr.iter()
314                    .filter_map(|block| block.get("text").and_then(|t| t.as_str()))
315                    .collect::<Vec<_>>()
316                    .join(" ")
317                    .into()
318            } else {
319                None
320            }
321        })
322        .unwrap_or_default()
323        .chars()
324        .take(500)
325        .collect()
326}

328/// Log a completed request to the database with accurate token counts and costs.
329#[allow(clippy::too_many_arguments)]
330async fn log_request(
331    db: &db::DbPool,
332    pricing_table: &pricing::PricingTable,
333    request_id: &str,
334    provider: &str,
335    model: &str,
336    original_model: &str,
337    was_substituted: bool,
338    usage: &normalizer::NormalizedUsage,
339    latency_ms: i64,
340    anomaly_multiplier: f64,
341    last_user_message: &str,
342) {
343    // Calculate costs using actual token counts from the API response
344    let costs = pricing::lookup(pricing_table, provider, model)
345        .map(|p| {
346            pricing::calculate_costs(
347                p,
348                usage.input_tokens,
349                usage.output_tokens,
350                usage.cache_read_tokens,
351                usage.cache_write_tokens,
352            )
353        })
354        .unwrap_or_else(|| {
355            tracing::warn!(
356                "No pricing data for {}/{}, costs will be zero",
357                provider,
358                model
359            );
360            pricing::CostBreakdown {
361                input_cost_usd: 0.0,
362                output_cost_usd: 0.0,
363                cache_read_cost_usd: 0.0,
364                cache_write_cost_usd: 0.0,
365                total_cost_usd: 0.0,
366            }
367        });
368
369    // Check for anomaly
370    let rolling_avg = db::read::get_rolling_avg_cost(db).unwrap_or(0.0);
371    let is_anomaly = rolling_avg > 0.0 && costs.total_cost_usd > (rolling_avg * anomaly_multiplier);
372    let anomaly_reason = if is_anomaly {
373        Some(format!(
374            "Cost ${:.4} is {:.1}x the 7-day average ${:.4}",
375            costs.total_cost_usd,
376            costs.total_cost_usd / rolling_avg,
377            rolling_avg
378        ))
379    } else {
380        None
381    };
382
383    // Classify task type from the user message
384    let task_type = classify_task(last_user_message);
385
386    let row = RequestRow {
387        id: request_id.to_string(),
388        timestamp: Utc::now().to_rfc3339(),
389        provider: provider.to_string(),
390        model: model.to_string(),
391        original_model: original_model.to_string(),
392        was_substituted,
393        input_tokens: usage.input_tokens,
394        output_tokens: usage.output_tokens,
395        cache_read_tokens: usage.cache_read_tokens,
396        cache_write_tokens: usage.cache_write_tokens,
397        input_cost_usd: costs.input_cost_usd,
398        output_cost_usd: costs.output_cost_usd,
399        cache_read_cost_usd: costs.cache_read_cost_usd,
400        cache_write_cost_usd: costs.cache_write_cost_usd,
401        total_cost_usd: costs.total_cost_usd,
402        latency_ms,
403        stop_reason: usage.stop_reason.clone(),
404        task_type: Some(task_type),
405        tags: None,
406        anomaly: is_anomaly,
407        anomaly_reason,
408    };
409
410    if let Err(e) = db::write::insert_request(db, &row) {
411        tracing::error!("Failed to log request {}: {}", request_id, e);
412    }
413}

415/// Simple keyword-based task classification.
416fn classify_task(message: &str) -> String {
417    let msg = message.to_lowercase();
418    if msg.contains("```")
419        || msg.contains("code")
420        || msg.contains("function")
421        || msg.contains("implement")
422        || msg.contains("debug")
423        || msg.contains("fix")
424    {
425        "code".to_string()
426    } else if msg.contains("summarize") || msg.contains("summary") || msg.contains("tldr") {
427        "summarization".to_string()
428    } else if msg.contains("translate") || msg.contains("translation") {
429        "translation".to_string()
430    } else if msg.contains("classify") || msg.contains("categorize") || msg.contains("label") {
431        "classification".to_string()
432    } else if msg.contains("extract") || msg.contains("parse") || msg.contains("json") {
433        "data_extraction".to_string()
434    } else if msg.contains("write")
435        || msg.contains("story")
436        || msg.contains("poem")
437        || msg.contains("creative")
438    {
439        "creative_writing".to_string()
440    } else if msg.contains("analyze") || msg.contains("analysis") || msg.contains("compare") {
441        "analysis".to_string()
442    } else if msg.contains("?")
443        || msg.contains("what")
444        || msg.contains("how")
445        || msg.contains("why")
446        || msg.contains("explain")
447    {
448        "question_answering".to_string()
449    } else {
450        "other".to_string()
451    }
