1use anyhow::Result;
2use reqwest::Client;
3use std::collections::HashMap;

5/// Forward a request to the real LLM API and return the raw response.
6pub async fn forward_request(
7    client: &Client,
8    target_url: &str,
9    method: &str,
10    headers: &HashMap<String, String>,
11    body: bytes::Bytes,
12) -> Result<reqwest::Response> {
13    let mut req_builder = match method {
14        "POST" => client.post(target_url),
15        "GET" => client.get(target_url),
16        "PUT" => client.put(target_url),
17        "DELETE" => client.delete(target_url),
18        "PATCH" => client.patch(target_url),
19        _ => client.post(target_url),
20    };
21
22    // Forward relevant headers
23    for (key, value) in headers {
24        let key_lower = key.to_lowercase();
25        // Skip hop-by-hop headers
26        if key_lower == "host" || key_lower == "connection" || key_lower == "content-length" {
27            continue;
28        }
29        req_builder = req_builder.header(key.as_str(), value.as_str());
30    }
31
32    // Send the request body
33    if !body.is_empty() {
34        req_builder = req_builder.body(body);
35    }
36
37    let response = req_builder.send().await?;
38
39    Ok(response)
40}

42/// Build the target URL from provider base URL and the original request path.
43pub fn build_target_url(base_url: &str, path: &str) -> String {
44    let base = base_url.trim_end_matches('/');
45    let path = path.trim_start_matches('/');
46    format!("{}/{}", base, path)
47}

49/// Check if a response is an SSE stream.
50pub fn is_sse_response(response: &reqwest::Response) -> bool {
51    response
52        .headers()
53        .get("content-type")
54        .and_then(|v| v.to_str().ok())
55        .map(|ct| ct.contains("text/event-stream"))
56        .unwrap_or(false)
57}
